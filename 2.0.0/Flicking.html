    Example
    var Flicking = tui.Flicking; // or require('tui-flicking');
var instance = new Flicking({
     element: document.getElementById('flick'),
     wrapper: document.getElementById('flick-panels'),
     flow: 'horizontal',
     circular: true,
     useMagnetic: true,
     effect: 'linear',
     flickRange: 100,
     duration: 300,
     itemClass: 'panel',
     itemTag: 'div',
     data: '<strong class="contents">panel</strong>'
});
